# Lagom Absorption Rate Engine 
 
This notebook calculates absorption rate and months of supply, filters to Lagom's target $200K-$300K price range and target states, summarizes markets at the CBSA/metro + price-band level, analyzes historical absorption trends, and builds a model to score high-potential markets. 
#%% 
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

from sklearn.model_selection import train_test_split
from sklearn.metrics import mean_absolute_error, mean_squared_error, r2_score
from sklearn.linear_model import LinearRegression
from sklearn.ensemble import RandomForestRegressor, GradientBoostingRegressor
from sklearn.tree import DecisionTreeRegressor
from sklearn.base import clone
from IPython.display import display

#%% md 
## 1. Load and merge source data 
#%% 
# Read in data sets
# These files are stored in the Lagom GCP bucket.
df_counties = pd.read_excel(
    "https://storage.googleapis.com/lagom-omsa/raw_data/US%20Counties.xlsx"
)

df_redfin = pd.read_csv(
    "https://storage.googleapis.com/lagom-omsa/raw_data/redfin_weekly_housing_market_data_most_recent.csv"
)

df_realtor = pd.read_csv(
    "https://storage.googleapis.com/lagom-omsa/raw_data/realtor_Inventory_Hotness_Metrics_County_History.csv"
)

# Keep only Redfin county-level records.
df_redfin_county = df_redfin[df_redfin['REGION_TYPE'] == 'county'].copy()

# Split Redfin region name into county and state.
df_redfin_county[['county', 'state']] = df_redfin_county['REGION_NAME'].str.extract(
    r'^(.*) County, ([A-Z]{2})$'
)

# Create a common county/state key for joining Redfin and Realtor.com data.
df_redfin_county['county_state'] = (
    df_redfin_county['county'].str.lower() + ', ' + df_redfin_county['state'].str.lower()
)

df_realtor['county_state'] = df_realtor['county_name'].str.lower()

# Convert Redfin weekly dates to YYYYMM month format so they can be joined to Realtor monthly data.
df_redfin_county['month_date_yyyymm'] = pd.to_datetime(
    df_redfin_county['PERIOD_BEGIN']
).dt.strftime('%Y%m')

df_redfin_county['month_date_yyyymm'] = df_redfin_county['month_date_yyyymm'].astype(str)
df_realtor['month_date_yyyymm'] = df_realtor['month_date_yyyymm'].astype(str)

# Aggregate Redfin weekly data to county-month level.
df_redfin_agg = df_redfin_county.groupby(
    ['county_state', 'month_date_yyyymm'],
    as_index=False
).mean(numeric_only=True)

# Merge Redfin and Realtor.com county-month data.
merged = pd.merge(
    df_redfin_agg,
    df_realtor,
    on=['county_state', 'month_date_yyyymm'],
    how='inner'
)

# Remove duplicate county-month records if any remain.
merged = merged.drop_duplicates(subset=['county_state', 'month_date_yyyymm'])

print(merged.shape)
print(merged[['county_state', 'month_date_yyyymm']].head(25))
#%% md 
## 2. Create absorption metrics and price bands 
#%% 
# Absorption rate measures how quickly available inventory is being sold.
merged['absorption_rate'] = (
    merged['ADJUSTED_AVERAGE_HOMES_SOLD'] /
    merged['ACTIVE_LISTINGS']
) * 100

# Months of supply estimates how many months it would take to sell current inventory.
merged['months_of_supply'] = (
    merged['ACTIVE_LISTINGS'] /
    merged['ADJUSTED_AVERAGE_HOMES_SOLD']
)

# Replace infinite values caused by divide-by-zero.
merged = merged.replace([np.inf, -np.inf], np.nan)

# Convert YYYYMM field into a real monthly date.
merged['month_date'] = pd.to_datetime(
    merged['month_date_yyyymm'].astype(str),
    format='%Y%m'
)

# Create price bands for Lagom's attainable housing analysis.
merged['price_band'] = pd.cut(
    merged['MEDIAN_SALE_PRICE'],
    bins=[0, 200000, 250000, 300000, 350000, float('inf')],
    labels=[
        '<$200K',
        '$200K-$250K',
        '$250K-$300K',
        '$300K-$350K',
        '$350K+'
    ],
    include_lowest=True
)

print(merged[['county_state', 'month_date', 'MEDIAN_SALE_PRICE', 'price_band', 'absorption_rate', 'months_of_supply']].head())
#%% md 
## 3. Filter to Lagom's target price band and target states 
#%% 
# Keep Lagom's target price range: $200K-$300K.
target_df = merged[
    merged['price_band'].isin([
        '$200K-$250K',
        '$250K-$300K'
    ])
].copy()

# Target states provided for the project.
target_states = [
    'al', 'ar', 'fl', 'ga', 'ky',
    'la', 'ms', 'mo', 'nc', 'sc',
    'tn', 'va', 'wv', 'ks', 'ok', 'tx'
]

# Extract state abbreviation from county_state, which looks like "county, st".
target_df['state'] = (
    target_df['county_state']
    .str.split(',')
    .str[-1]
    .str.strip()
    .str.lower()
)

# Filter to target geography.
target_df = target_df[
    target_df['state'].isin(target_states)
].copy()

print(target_df.shape)
print(target_df[['county_state', 'state', 'price_band']].head())
#%% md 
## 4. Aggregate to CBSA/metro + price-band + month level 
 
CBSA is used as the primary market level because it represents metro housing markets. County is retained for drill-down analysis later. 
#%% 
# Aggregate county rows up to CBSA/metro + price band + month.
# For absorption and months of supply, calculate from summed homes sold and summed active listings.
cbsa_monthly = target_df.groupby(
    ['cbsa_title', 'price_band', 'month_date'],
    observed=True
).agg(
    states=('state', lambda x: ', '.join(sorted(x.dropna().unique()))),
    avg_median_sale_price=('MEDIAN_SALE_PRICE', 'mean'),
    active_listings=('ACTIVE_LISTINGS', 'sum'),
    homes_sold=('ADJUSTED_AVERAGE_HOMES_SOLD', 'sum'),
    median_days_on_market=('MEDIAN_DAYS_ON_MARKET', 'mean'),
    hotness_score=('hotness_score', 'mean'),
    demand_score=('demand_score', 'mean'),
    supply_score=('supply_score', 'mean')
).reset_index()

cbsa_monthly['absorption_rate'] = (
    cbsa_monthly['homes_sold'] /
    cbsa_monthly['active_listings']
) * 100

cbsa_monthly['months_of_supply'] = (
    cbsa_monthly['active_listings'] /
    cbsa_monthly['homes_sold']
)

cbsa_monthly = cbsa_monthly.replace([np.inf, -np.inf], np.nan)

print(cbsa_monthly.head())
#%% md 
## 5. CBSA + price-band summary 
#%% 
cbsa_price_summary = cbsa_monthly.groupby(
    ['cbsa_title', 'price_band'],
    observed=True
).agg(
    states=('states', 'first'),
    avg_absorption_rate=('absorption_rate', 'mean'),
    avg_months_of_supply=('months_of_supply', 'mean'),
    avg_sale_price=('avg_median_sale_price', 'mean'),
    avg_active_listings=('active_listings', 'mean'),
    avg_homes_sold=('homes_sold', 'mean'),
    latest_month=('month_date', 'max')
).reset_index()

print(cbsa_price_summary.head())

# Need to sort by avg_absorption_rate desc
#%% md 
## 6. Fastest-moving markets today 
#%% 
latest_month = cbsa_monthly['month_date'].max()

latest_markets = cbsa_monthly[
    cbsa_monthly['month_date'] == latest_month
].copy()

top_current_markets = latest_markets.sort_values(
    by='absorption_rate',
    ascending=False
)

print("Top Current Markets in $200K-$300K Band")
print(top_current_markets[[
    'cbsa_title',
    'states',
    'price_band',
    'avg_median_sale_price',
    'active_listings',
    'homes_sold',
    'absorption_rate',
    'months_of_supply'
]].head(20))
#%% md 
## 7. Rolling-window trend analysis 
 
A 12-month average captures recent market performance, while a 24-month average captures longer-term performance. Absorption velocity is defined as the 12-month average minus the 24-month average. Positive values suggest recent absorption is stronger than the longer-term baseline. 
#%% 
cbsa_monthly = cbsa_monthly.sort_values(
    by=['cbsa_title', 'price_band', 'month_date']
)

cbsa_monthly['absorption_12mo_avg'] = cbsa_monthly.groupby(
    ['cbsa_title', 'price_band'],
    observed=True
)['absorption_rate'].transform(
    lambda x: x.rolling(window=12, min_periods=6).mean()
)

cbsa_monthly['absorption_24mo_avg'] = cbsa_monthly.groupby(
    ['cbsa_title', 'price_band'],
    observed=True
)['absorption_rate'].transform(
    lambda x: x.rolling(window=24, min_periods=12).mean()
)

cbsa_monthly['absorption_velocity'] = (
    cbsa_monthly['absorption_12mo_avg'] -
    cbsa_monthly['absorption_24mo_avg']
)

print(cbsa_monthly[[
    'cbsa_title',
    'price_band',
    'month_date',
    'absorption_rate',
    'absorption_12mo_avg',
    'absorption_24mo_avg',
    'absorption_velocity'
]].tail())
#%% 
# =====================================================
# CREATE FUTURE ABSORPTION TARGETS
# =====================================================

cbsa_monthly = cbsa_monthly.sort_values(
    by=['cbsa_title', 'price_band', 'month_date']
)

forecast_horizons = {
    '2 Years': 24,
    '3 Years': 36,
    '4 Years': 48,
    '5 Years': 60
}

for label, months in forecast_horizons.items():
    cbsa_monthly[f'future_absorption_{months}m'] = (
        cbsa_monthly
        .groupby(['cbsa_title', 'price_band'])['absorption_rate']
        .shift(-months)
    )
#%% md 
## 8. Top markets by absorption velocity 
#%% 
latest_velocity = cbsa_monthly[
    cbsa_monthly['month_date'] == latest_month
].copy()

top_velocity_markets = latest_velocity.sort_values(
    by='absorption_velocity',
    ascending=False
)

print("Top Markets by Absorption Velocity")
print(top_velocity_markets[[
    'cbsa_title',
    'states',
    'price_band',
    'avg_median_sale_price',
    'absorption_rate',
    'absorption_12mo_avg',
    'absorption_24mo_avg',
    'absorption_velocity',
    'months_of_supply'
]].head(20))
#%% md 
## 9. Historical absorption trend charts 
#%% 
top_market_rows = top_current_markets.head(5)

for _, row in top_market_rows.iterrows():

    cbsa = row['cbsa_title']
    band = row['price_band']

    market_data = cbsa_monthly[
        (cbsa_monthly['cbsa_title'] == cbsa) &
        (cbsa_monthly['price_band'] == band)
    ].copy()

    market_data = market_data.sort_values('month_date')

    plt.figure(figsize=(10, 5))

    plt.plot(

        market_data['absorption_rate'],
        marker='o',
        label='Monthly Absorption'
    )

    plt.plot(
        market_data['month_date'],
        market_data['absorption_12mo_avg'],
        marker='o',

    )

    plt.title(f'Absorption Trend: {cbsa} | {band}')
    plt.xlabel('Month')
    plt.ylabel('Absorption Rate')
    plt.legend()
    plt.tight_layout()
    plt.show()
#%% md 
## 10. Forecasting model: 2, 3, 4, and 5 years ahead 
 
This section creates true future absorption forecasts by training models where the target is absorption rate 24, 36, 48, or 60 months after the current observation. 
 
#%% 
# =====================================================
# MODELING FEATURES AND MODEL CANDIDATES
# =====================================================

# These features describe the market at the current point in time.
# The target created later is the absorption rate 24, 36, 48, or 60 months in the future.
features = [
    'avg_median_sale_price',
    'median_days_on_market',
    'hotness_score',
    'demand_score',
    'supply_score',
    'months_of_supply',
    'absorption_12mo_avg',
    'absorption_24mo_avg',
    'absorption_velocity'
]

models = {
    'Linear Regression': LinearRegression(),

    'Decision Tree': DecisionTreeRegressor(
        max_depth=10,
        random_state=42
    ),

    'Random Forest': RandomForestRegressor(
        n_estimators=200,
        random_state=42
    ),

    'Gradient Boosting': GradientBoostingRegressor(
        n_estimators=200,
        random_state=42
    )
}

#%% 
# =====================================================
# TRAIN MODELS FOR 2, 3, 4, AND 5 YEAR FORECASTS
# =====================================================

forecast_targets = {
    '2 Years': 'future_absorption_24m',
    '3 Years': 'future_absorption_36m',
    '4 Years': 'future_absorption_48m',
    '5 Years': 'future_absorption_60m'
}

all_model_results = []
all_market_predictions = {}
best_models_by_horizon = {}
feature_importance_by_horizon = {}

# Use the latest available market observations as the rows to forecast from.
latest_month = cbsa_monthly['month_date'].max()
latest_scoring_df = cbsa_monthly[
    cbsa_monthly['month_date'] == latest_month
].copy()

for horizon_label, target in forecast_targets.items():

    # These are descriptive columns we want to keep with the output.
    # Do NOT repeat any columns that are already in the features list.
    output_cols = [
        'cbsa_title',
        'states',
        'price_band',
        'month_date',
        'absorption_rate'
    ]

    # Training rows must have current features and the future absorption target.
    model_df = cbsa_monthly[
        features + [target] + output_cols
    ].dropna().copy()

    model_df = model_df.sort_values('month_date')

    X = model_df[features]
    y = model_df[target]

    # Time-based split: train on older observations, test on newer observations.
    split_index = int(len(model_df) * 0.80)

    X_train = X.iloc[:split_index]
    X_test = X.iloc[split_index:]
    y_train = y.iloc[:split_index]
    y_test = y.iloc[split_index:]

    horizon_results = []
    fitted_models = {}

    for name, base_model in models.items():

        model = clone(base_model)
        model.fit(X_train, y_train)

        preds = model.predict(X_test)

        horizon_results.append({
            'Forecast_Horizon': horizon_label,
            'Target_Column': target,
            'Model': name,
            'Train_Rows': len(X_train),
            'Test_Rows': len(X_test),
            'MAE': mean_absolute_error(y_test, preds),
            'RMSE': np.sqrt(mean_squared_error(y_test, preds)),
            'R2': r2_score(y_test, preds)
        })

        fitted_models[name] = model

    horizon_results_df = pd.DataFrame(horizon_results).sort_values(
        by='R2',
        ascending=False
    )

    best_model_name = horizon_results_df.iloc[0]['Model']

    # Refit the best model on all historical rows with known future outcomes.
    best_model = clone(models[best_model_name])
    best_model.fit(X, y)

    best_models_by_horizon[horizon_label] = {
        'model_name': best_model_name,
        'model': best_model,
        'target': target
    }

    # Score the latest month to forecast future absorption from today's/latest conditions.
    scoring_df = latest_scoring_df[
        features + output_cols
    ].dropna().copy()

    prediction_col = f'predicted_absorption_{horizon_label}'

    scoring_df[prediction_col] = best_model.predict(scoring_df[features])
    scoring_df['Forecast_Horizon'] = horizon_label
    scoring_df['Best_Model'] = best_model_name

    market_predictions = scoring_df[
        [
            'Forecast_Horizon',
            'Best_Model',
            'cbsa_title',
            'states',
            'price_band',
            'month_date',
            'absorption_rate'
        ] + features + [prediction_col]
    ].copy()

    market_predictions = market_predictions.sort_values(
        by=prediction_col,
        ascending=False
    )

    all_model_results.extend(horizon_results)
    all_market_predictions[horizon_label] = market_predictions

    # Store feature importance if the best model supports it.
    if hasattr(best_model, 'feature_importances_'):
        importance = pd.DataFrame({
            'Forecast_Horizon': horizon_label,
            'Best_Model': best_model_name,
            'Feature': features,
            'Importance': best_model.feature_importances_
        }).sort_values(
            by='Importance',
            ascending=False
        )

        feature_importance_by_horizon[horizon_label] = importance

#%% md
